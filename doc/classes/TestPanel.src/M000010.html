  def test_panel_creation_out_of_boundaries
    panel = Bio::Graphics::Panel.new(1000, :width => 500, :clickable => false,
                                     :display_range => -7..5000)
    assert_same(0, panel.display_start)
    assert_same(1000, panel.display_stop)
  end
