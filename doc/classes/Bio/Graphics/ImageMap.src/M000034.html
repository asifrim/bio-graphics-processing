  def to_s
    output = Array.new
    output.push('<map name="image_map" border="1">')
    @elements.each do |e|
      area = e.to_s
      if area != ''
        output.push(e.to_s)
      end
    end
    output.push('</map>')
    return output.join("\n")
  end
